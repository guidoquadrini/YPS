 45:         static $do_setup = TRUE;

 47:         $r = '';

 49:         if ($alias != '' && ! is_array($alias))
 50:         {
 51:             $alias = array($alias => $field_id);
 52:         }
 53: 
 54:         if ($do_setup === TRUE)

 56:                 $do_setup = FALSE;
